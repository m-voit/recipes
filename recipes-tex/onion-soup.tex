\begin{recipe}{Zwiebelsuppe}{\faIcon{utensils}}{30min \faIcon{user-clock}, 30min \faIcon{user-cog}}

  \freeform{}\textit{}

  \ing[1]{kg}{Zwiebeln}
  \ing[5]{EL}{Öl}
  \ing[2]{EL}{Mehl}
  \ing[2]{TL}{Zucker}
  \ing[]{}{Salz}
  \ing[]{}{Pfeffer}

  \textbf{10min}
  Zwiebeln halbieren und in Ringe schneiden.
  Zwiebeln im Öl glasig braten und anschließend mit Mehl und Zucker bestäuben.
  Mit etwas Salz und Pfeffer würzen.

  \ing[300]{ml}{Weißwein}
  \ing[1]{l}{Gemüsebrühe}
  \ing[1]{x}{Mehl}

  \textbf{15min}
  Mit dem Wein nach und nach ablöschen.
  Dann die Brühe dazugeben und ca. 15min köcheln lassen.

  \ing[50]{g}{Butter}
  \ing[1]{x}{Baguette}
  \ing[1]{x}{Maisstärke}

  \textbf{5 min}
  Butter in der Pfanne erhitzen und Baguette goldbraun braten.

  \ing[8]{EL}{Bergkäse}
  \ing[3]{EL}{Parmesan}
  \ing[1]{TL}{Paprikapulver}

  \textbf{30 min}
  Suppe in ofenfeste Schalen füllen, das Baguette auf die Suppe legen, mit Käse und Paprika bestreuen und goldbraun backen.

  \freeform{}\hrulefill{}

  \freeform{}\faIcon{lightbulb}
  Mit selbstgemachter Gemüsebrühe zubereiten.

\end{recipe}
