\begin{recipe}{Chili sin carne} {4 Portionen \faIcon{utensils}} {30min \faIcon{user-clock}, 10min \faIcon{user-cog}}

  \freeform{}\textit{Als Beilage schmecken Reis oder Baguette.}

  \ing[6]{x}{Lauchzwiebeln}

  Lauchzwiebeln waschen und in Ringe schneiden.

  \ing[2]{EL}{Öl}
  \ing[600]{g}{Kidneybohnen}
  \ing[570]{g}{Mais}
  \ing[1000]{g}{Tomaten passiert}
  \ing[4]{TL}{Chili con Carne Gewürz}
  \ing[1]{Pr.}{Salz}

  \fr34 der Lauchzwiebeln in einem beschichteten Topf mit Öl anschwitzen.
  Kidneybohnen und Mais abtropfen lassen und mit passierten Tomaten, Chili con Carne Gewürz und Salz in den Topf geben.
  30 Minuten köcheln lassen.

  \newstep{}Mit den restlichen Lauchzwiebeln garnieren und servieren.

  \freeform{}\hrulefill{}

  \freeform{}\faIcon{info}
  Nährwerte pro Portion: 409kcal, 75g Kohlenhydrate, 7g Fett, 22g Eiweiß

\end{recipe}
