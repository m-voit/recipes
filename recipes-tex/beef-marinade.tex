\begin{recipe}{Rindfleischmarinade}{400g \faIcon{utensils}}{12h \faIcon{user-clock}, 5min \faIcon{user-cog}}

  \freeform{}\textit{Marinade für Rindfleisch.}

  \ingredient[400]{g}{Rind}
  \ingredient[4]{EL}{Olivenöl raffiniert}
  \ingredient[2]{TL}{Gewürze}
  \ingredient[\fr12]{TL}{Pfeffer schwarz}
  \ingredient[1]{x}{Knoblauchzehe gehackt}
  \ingredient[1]{x}{kleine Zwiebel, gehackt}
  \ingredient[4]{g}{Salz}

  \textbf{5min}
  Alle Zutaten in einer Schüssel vermischen, kurz ziehen lassen und anschließend das Fleisch damit einreiben.

  \newstep{}\textbf{12h}
  Die Marinade ins Fleisch einmassieren und danach über Nacht im Kühlschrank ruhen lassen.

  \freeform{}\hrulefill{}

  \freeform{}\faIcon{lightbulb}
  Das marinierte Fleisch in einem Tiefkühlbeutel luftdicht verschlossen lagern.

\end{recipe}
