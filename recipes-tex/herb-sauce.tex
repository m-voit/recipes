\begin{recipe}{Kräuter-Sauce}{\faIcon{utensils}}{12h \faIcon{user-clock}, 10min\faIcon{user-cog}}

  \freeform{}\textit{Ergibt eine Kräuter-Sauce.}

  \ing[1]{Bund}{Petersilie}
  \ing[\fr12]{Bund}{Koriander}
  \ing[1]{x}{Knoblauchzehe}
  \ing[1]{x}{Chilischote, rot}
  \ing[\fr12]{Zitrone}{Saft}
  \ing[1]{EL}{Ahornsirup}
  \ing[2]{TL}{Tahin}
  \ing[1\fr12]{EL}{Olivenöl}
  \ing[150]{ml}{Apfelsaft}
  \ing[1]{Msp.}{Kardamonpulver}
  \ing[]{}{Salz}
  \ing[]{}{Pfeffer}

  \textbf{10min}
  Zuerst die Petersilie und den Koriander hacken.
  Danach Knoblauchzehe schälen und Chilischote gründlich hacken.
  Saft der Zitrone auspressen und mit Ahornsirup, Tahin und Olivenöl vermischen.
  Die Kräuter unterheben.
  Eventuell etwas Wasser zugeben.
  Mit Kardamompulver, Salz und Pfeffer abschmecken.
  In eine saubere Flasche füllen.

  \newstep{}\textbf{12h}
  Über Nacht im Kühlschrank ziehen lassen.

  \freeform{}\hrulefill{}

\end{recipe}
