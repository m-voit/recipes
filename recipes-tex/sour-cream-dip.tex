\begin{recipe}{Sour Cream}{4 Personen \faIcon{utensils}}{12h \faIcon{user-clock}, 10min \faIcon{user-cog}}

  \freeform{}\textit{Ideal als Dip zu Ofenkartoffeln.}

  \ing[150]{ml}{Schmand}
  \ing[150]{ml}{Saure Sahne}
  \ing[1]{x}{kleine Zwiebel}
  \ing[2]{x}{Knoblauchzehe}
  \ing[1]{Bund}{Schnittlauch}

  \textbf{10min}
  Zwiebel und Knoblauchzehe klein hacken.
  Anschließend alle Zutaten in einer Schale vermengen.

  \ing[]{}{Salz}
  \ing[]{}{Pfeffer}

  Mit Salz und Pfeffer abschmecken.

  \newstep{}\textbf{12h}
  Danach die Sour Cream im Kühlschrank über Nacht ziehen lassen.

  \freeform{}\hrulefill{}

\end{recipe}
