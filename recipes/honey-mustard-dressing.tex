\begin{recipe}{Honig-Senf-Dressing} {0ml} {0min \faIcon[regular]{clock}, 0min \faIcon[regular]{user}}

  \freeform{}\textit{Ergibt ein würziges Salatdressing.}

  \ing[150]{g}{Balsamico, weiß}
  \ing[120]{g}{Honig}
  \ing[100]{ml}{Wasser}
  \ing[3]{EL}{Dijonsenf}
  \ing[2]{TL}{Meersalz}

  Alle Zutaten mit einem Schneebesen verrühren.

  \ing[150]{g}{Olivenöl, extra nativ}

  Danach das Öl hinzugeben und alles nochmal kräftig mit dem Schneebesen aufschlagen.

  \ing[]{}{Pfeffer, schwarz}

  Zum Schluss mit schwarzem Pfeffer abschmecken.

  \freeform{}\hrulefill{}

  \freeform{}\faIcon[regular]{lightbulb}
  Hält sich im Kühlschrank ca. 2 Wochen.

\end{recipe}
